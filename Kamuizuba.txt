-- Tạo hiệu ứng Kamui
local function CreateKamuiEffect()
    local Effect = Instance.new("Part")
    Effect.Size = Vector3.new(5, 5, 5)
    Effect.Position = Plr.Character.HumanoidRootPart.Position
    Effect.Anchored = true
    Effect.CanCollide = false
    Effect.Transparency = 1
    Effect.Parent = Workspace

    local Mesh = Instance.new("SpecialMesh", Effect)
    Mesh.MeshId = KamuiEffect
    Mesh.TextureId = KamuiEffect
    Mesh.Scale = Vector3.new(5, 5, 5)

    game:GetService("Debris"):AddItem(Effect, 2) -- Xóa hiệu ứng sau 2 giây
end

-- Nút Toggle
Toggle.MouseButton1Click:Connect(function()
    if not Plr.Character or not Plr.Character:FindFirstChild("HumanoidRootPart") then return end
    
    if Status.Text == "off" then
        Clipon = true
        Status.Text = "on"
        Status.TextColor3 = Color3.new(0, 1, 0)

        -- Tạo hiệu ứng Kamui khi bật
        CreateKamuiEffect()

        -- Kích hoạt noclip
        Stepped = RunService.Stepped:Connect(function()
            if Clipon then
                for _, v in pairs(Plr.Character:GetChildren()) do
                    if v:IsA("BasePart") then
                        v.CanCollide = false
                    end
                end
            else
                Stepped:Disconnect()
            end
        end)

        task.wait(2) -- Chờ hiệu ứng 2s
    else
        Clipon = false
        Status.Text = "off"
        Status.TextColor3 = Color3.new(1, 0, 0)

        -- Tạo hiệu ứng Kamui khi tắt
        CreateKamuiEffect()
        task.wait(2) -- Chờ hiệu ứng 2s
    end
end)
